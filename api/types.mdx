## Overview

Moving Icons is built with TypeScript and exports comprehensive type definitions for full type safety in your Svelte applications.

## IconProps

The main interface defining all props accepted by icon components.

<CodeGroup>
```typescript Interface Definition
export interface IconProps {
  color?: string;
  size?: number;
  strokeWidth?: number;
  animate?: boolean;
  class?: string;
}
```

```typescript Import
import type { IconProps } from 'moving-icons';
```
</CodeGroup>

### Properties

<ResponseField name="color" type="string" optional>
  The stroke color of the icon. Accepts any valid CSS color value.
  
  **Default**: `'currentColor'`
</ResponseField>

<ResponseField name="size" type="number" optional>
  The width and height of the icon in pixels.
  
  **Default**: `24`
</ResponseField>

<ResponseField name="strokeWidth" type="number" optional>
  The width of the icon's stroke lines in pixels.
  
  **Default**: `2`
</ResponseField>

<ResponseField name="animate" type="boolean" optional>
  Controls whether the icon is currently animating.
  
  **Default**: `false`
</ResponseField>

<ResponseField name="class" type="string" optional>
  Custom CSS class name(s) to apply to the icon's wrapper element.
  
  **Default**: `''`
</ResponseField>

## Complete Example

<CodeGroup>
```typescript Svelte Component with Types
<script lang="ts">
  import { Heart, Star, ThumbsUp } from 'moving-icons';
  import type { IconProps, IconName } from 'moving-icons';

  interface IconButtonProps {
    icon: IconName;
    label: string;
    iconProps?: IconProps;
    onClick?: () => void;
  }

  let { icon, label, iconProps, onClick }: IconButtonProps = $props();

  const defaultIconProps: IconProps = {
    size: 20,
    strokeWidth: 2,
    color: 'currentColor'
  };

  const mergedProps = { ...defaultIconProps, ...iconProps };
</script>

<button onclick={onClick}>
  {#if icon === 'heart'}
    <Heart {...mergedProps} />
  {:else if icon === 'star'}
    <Star {...mergedProps} />
  {:else if icon === 'thumbs-up'}
    <ThumbsUp {...mergedProps} />
  {/if}
  <span>{label}</span>
</button>
```

```typescript Type-Safe Icon Mapper
import type { IconName } from 'moving-icons';
import * as Icons from 'moving-icons';

type IconComponent = typeof Icons[keyof typeof Icons];

function getIconComponent(name: IconName): IconComponent | undefined {
  // Convert kebab-case to PascalCase
  const componentName = name
    .split('-')
    .map(part => part.charAt(0).toUpperCase() + part.slice(1))
    .join('');
  
  return Icons[componentName as keyof typeof Icons] as IconComponent;
}

const HeartIcon = getIconComponent('heart');
```
</CodeGroup>
